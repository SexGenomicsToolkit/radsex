26 double get_chi_squared_p(double chi_squared) {
27 
28  /* p is given by 1 - P(chi_squared, df) where P is the Cumulative Distribution Function of the Chi-squared distribution.
29  * P is also the regularized gamma function. P is computed using samtool's implementation of the regularized gamma function.
30  * Source: https://en.wikipedia.org/wiki/Chi-squared_distribution#Cumulative_distribution_function
31  * DF is always 1 in our case
32  */
33 
34  return std::min(1.0, 1 - kf_gammap(0.5, chi_squared / 2));
35 }

41 double get_chi_squared(uint n_group1, uint n_group2, uint total_group1, uint total_group2) {
42 
43  /* Chi squared is computed from the number of individuals in each group.
44  * Yates correction is applied and the shortcut formula for 2x2 table is used.
45  * Source: https://en.wikipedia.org/wiki/Yates%27s_correction_for_continuity
46  */
47 
48  uint N = total_group1 + total_group2;
49  uint Ns = total_group1, Nf = total_group2;
50  uint Na = n_group1 + n_group2, Nb = total_group1 + total_group2 - n_group1 - n_group2;
51 
52  int temp = static_cast<int>((n_group1 * total_group2) - (n_group2 * total_group1));
53  temp = std::abs(temp);
54  double temp2 = std::max(0.0, double(temp) - N/2);
55  temp2 *= temp2;
56 
57  return N * temp2 / Ns / Nf / Na / Nb;
58 }

71
