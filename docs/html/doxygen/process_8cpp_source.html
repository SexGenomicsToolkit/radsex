28 std::vector<InputFile> get_input_files(const std::string& input_dir_path) {
29 
30  /* Gets all relevant files (from supported formats) from the input directory.
31  * Files are stored as InputFile object which contains all relevant information about the file.
32  * The function returns a vector of InputFiles.
33  */
34 
35  DIR* dir = opendir(input_dir_path.c_str());
36 
37  if(!dir) {
38  log("Could not read from the directory <" + input_dir_path + ">", LOG_ERROR);
39  exit(0);
40  }
41 
42  struct dirent* dir_content;
43  std::vector<InputFile> files;
44  std::string current_file, extension, individual_name;
45  std::vector<std::string> split_name;
46  InputFile temp;
47 
48  while ((dir_content = readdir(dir))) {
49 
50  current_file = dir_content->d_name;
51  split_name = split(current_file, ".");
52  size_t s = split_name.size();
53  extension = "";
54  individual_name = split_name[0];
55 
56  // Get file name and extension (even when there is "." in the file name)
57  if (s > 1) {
58  if (split_name[s - 1] == "gz" and s > 2) {
59  extension = "." + split_name[s - 2] + "." + split_name[s - 1];
60  for (uint i=1; i<split_name.size() - 2; ++i) individual_name += "." + split_name[i];
61  } else {
62  extension = "." + split_name[s - 1];
63  for (uint i=1; i<split_name.size() - 1; ++i) individual_name += "." + split_name[i];
64  }
65  }
66 
67  if(std::find(extensions.begin(), extensions.end(), extension) != extensions.end()) {
68  temp.individual_name = individual_name;
69  temp.path = input_dir_path + current_file;
70  temp.extension = extension;
71  temp.processed = false;
72  files.push_back(temp);
73  }
74  }
75 
76  if (files.size() == 0) {
77 
78  log("No valid input file found in input directory \"" + input_dir_path + ">");
79  std::string valid_formats_message = "Input files are detected based on extensions <";
80  for (auto& extension: extensions) valid_formats_message += extension;
81  if (extension != extensions.back()) valid_formats_message += ", ";
82  valid_formats_message += ">";
83  log(valid_formats_message);
84  exit(1);
85 
86  }
87 
88  log("Found <" + std::to_string(files.size()) + "> reads files");
89 
90  return files;
91 }

97 void process(Parameters& parameters) {
98 
99  /* The process_reads function does the following:
100  * - retrieve all sequence files from the input directory
101  * - create file processors according to the value of n_threads (allows to process files in parallel)
102  * - once all files are processed, output the results
103  */
104 
105  std::chrono::steady_clock::time_point t_begin = std::chrono::steady_clock::now();
106  log("RADSex <process> started");
107 
108  if (parameters.input_dir_path.back() != '/') parameters.input_dir_path += "/"; // Append "/" to the end of the path if it's missing
109 
110  std::vector<InputFile> input_files = get_input_files(parameters.input_dir_path);
111 
112  std::unordered_map<std::string, std::unordered_map<std::string, uint16_t>> results;
113 
114  std::vector<std::thread> threads;
115  std::mutex results_mutex, files_mutex;
116 
117  // Create and run all file processors
118  for (uint i=0; i<parameters.n_threads; ++i) {
119  threads.push_back(std::thread(file_processor, std::ref(input_files), std::ref(results), std::ref(results_mutex), std::ref(files_mutex)));
120  }
121 
122  for (auto &t : threads) t.join();
123 
124  // Create list of individuals
125  std::vector<std::string> individuals;
126  for (auto i: input_files) individuals.push_back(i.individual_name);
127 
128  std::ofstream output_file = open_output(parameters.output_file_path);
129 
130  output_file << "#Number of markers : " << results.size() << "\n";
131  output_file << "id\tsequence";
132  for (auto& i: individuals) output_file << "\t" << i;
133  output_file << "\n";
134 
135  log("Writing marker depths to output file");
136  uint id = 0;
137  bool print = true;
138 
139  uint marker_processed_tick = static_cast<uint>(results.size() / 100);
140  uint64_t n_processed_markers = 0;
141 
142  // Fill line by line
143  for (auto marker: results) {
144 
145  if (parameters.min_depth > 1) {
146  print = false;
147  for (auto i: marker.second) {
148  if (i.second >= parameters.min_depth) {
149  print = true;
150  break;
151  }
152  }
153  }
154 
155  if (print) {
156  output_file << id << "\t" << marker.first;
157  for (auto i: individuals) output_file << "\t" << marker.second[i];
158  output_file << "\n";
159  ++id;
160  }
161 
162  log_progress_bar(n_processed_markers, marker_processed_tick);
163 
164  }
165 
166  log("RADSex process ended (total runtime: " + get_runtime(t_begin) + ")");
167 }

173 inline void file_processor(std::vector<InputFile>& input_files, std::unordered_map<std::string, std::unordered_map<std::string, uint16_t>>& results, std::mutex& results_mutex, std::mutex& files_mutex) {
174 
175  /* The file processor looks for files to process and process them
176  * It is meant to be used in parallel to increase throughput
177  * A mutex protects the list of input files and another mutex protects the results structure
178  */
179 
180 
181  bool remaining_files = true;
182 
183  while (remaining_files) {
184 
185  files_mutex.lock();
186 
187  for (std::vector<InputFile>::iterator it = input_files.begin(); it != input_files.end(); ++it) {
188 
189  if (not it->processed) {
190 
191  it->processed = true;
192  remaining_files = true;
193  files_mutex.unlock();
194  process_file(*it, results, results_mutex);
195  break;
196 
197  } else {
198 
199  remaining_files = false;
200 
201  }
202  }
203 
204  files_mutex.unlock();
205 
206  }
207 }

213 inline void process_file(InputFile& input_file, std::unordered_map<std::string, std::unordered_map<std::string, uint16_t>>& results, std::mutex& results_mutex) {
214 
215  /* This function uses the Kseq generic file stream from Heng Li to read input files.
216  * The coverage of each sequence is computed without allowing for any mismatches.
217  * This is basically a read counter.
218  */
219 
220  std::unordered_map<std::string, uint16_t> temp_results; // Temporary result storage to avoid constantly locking/unlocking the mutex
221  gzFile file;
222  kseq_t* sequence; // Create the seq object
223  int line_n; // Stores the length of a line
224 
225  file = gzopen(input_file.path.c_str(), "r");
226 
227  if (not file) {
228 
229  log("Could not open reads file <" + input_file.path + ">", LOG_ERROR);
230  exit(1);
231 
232  }
233 
234  sequence = kseq_init(file); // Initialize the seq object
235 
236  // Read through the file and store the results
237  while ((line_n = kseq_read(sequence)) >= 0) ++temp_results[sequence->seq.s];
238 
239  kseq_destroy(sequence); // Destroy the seq object
240  gzclose(file);
241 
242  // Transfer the results from the temp data structure to the full data structure
243  results_mutex.lock();
244  for (auto marker : temp_results) results[marker.first][input_file.individual_name] += marker.second;
245  results_mutex.unlock();
246 
247  log("Finished processing individual <" + input_file.individual_name + ">");
248 }
