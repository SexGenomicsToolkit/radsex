184 inline void log_progress_bar(uint64_t& n_processed_markers, const uint32_t marker_processed_tick, char symbol = '#', uint16_t ticks = 50) {
185 
186  uint tick_size = 100 / ticks;
187 
188  if (++n_processed_markers % (tick_size * marker_processed_tick) != 0) return; // Not a step
189 
190  uint16_t progress = static_cast<uint16_t>(n_processed_markers / (tick_size * marker_processed_tick));
191 
192  std::string bar = "Progress: [";
193 
194  for(uint i=0; i < ticks; ++i) {
195  i < progress ? bar += symbol : bar += ' ';
196  }
197 
198  bar += "] - " + std::to_string(progress * tick_size) + "% (" + std::to_string(n_processed_markers) + " markers)";
199 
200  progress < ticks ? bar += '\r' : bar += '\n';
201 
202  log(bar, LOG_INFO, false);
203 
204 }
